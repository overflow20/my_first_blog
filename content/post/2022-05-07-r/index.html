kurtosis <- function(x, na.rm = FALSE) {
  # 去掉缺失值
  if (na.rm) x <- x[!is.na(x)]
  return(sum((x - mean(x))^4) / (length(x) * var(x)^2) - 3)
}#该函数有两个参数，数据向量x和是否删除缺失值 na.rm，后者有默认值 FALSE；
